>这题题目很长，不过很简单，男校友、女校友只是干扰，这题用一个flag就可以搞定，如果出现一个不到场的情况，当个校友的flag直接置为否就可以了
>上代码

```c++
#include <iostream>
#include <string>
using namespace std;

int flag[102];

int main()
{
	int N,D,i;
	int tmp;
	while (cin>>N>>D)
	{
		if(0 == N && 0== D)
			break;

		memset(flag,0,sizeof(flag));
		for(i=0;i<D;++i)
			for(int j=0;j<N;++j)
			{
				cin>>tmp;
				if(0 == tmp)
					flag[j]=1;
			}

			for (i=0;i<N;++i)
			{
				if(0 == flag[i])
				{
					cout<<"yes"<<endl;
					break;
				}
			}
			if(i==N)
				cout<<"no"<<endl;
	}

	return 0;
}

```
